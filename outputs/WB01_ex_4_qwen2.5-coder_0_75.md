```py
# Variables to store the input data
distance_travelled = 1000  # Distance in kilometers
litres_of_diesel_used = 85  # Liters of diesel used
cost_per_litre_of_diesel = 2.5  # Cost per liter of diesel
other_costs = 300  # Other costs (maintenance, etc.)

# Calculate the kilometres travelled per litre
kilometres_travelled_per_litre = distance_travelled / litres_of_diesel_used

# Calculate the total cost of the trip
total_cost_of_trip = (litres_of_diesel_used * cost_per_litre_of_diesel) + other_costs

# Calculate the total cost per kilometre
total_cost_per_kilometre = total_cost_of_trip / distance_travelled

# Print the results
print("Kilometres travelled per litre:", kilometres_travelled_per_litre)
print("Total cost of the trip:", total_cost_of_trip)
print("Total cost per kilometre:", total_cost_per_kilometre)
```
